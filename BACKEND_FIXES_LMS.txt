2. QUIZ QUESTION DATA FIX (CreateQuizQuestions)
-----------------------------------------------
Fixed the OptionC/OptionD copy-paste error and added the CorrectAnswer assignment.

[HttpPost("quiz-questions")]
[Authorize]
public async Task<IActionResult> CreateQuizQuestions(CreateModuleQuizQuestionDto dto)
{
    var module = await _context.CourseModules.FindAsync(dto.ModuleId);
    if (module == null)
        return NotFound(new { message = "Module not found." });

    var validAnswer = new[] { "A", "B", "C", "D" };
    if (!validAnswer.Contains(dto.CorrectAnswer.ToUpper()))
        return BadRequest(new { message = "Correct answer must be A, B, C or D" });

    var question = new ModuleQuizQuestion
    {
        ModuleId = dto.ModuleId,
        Question = dto.Question,
        optionA = dto.OptionA,
        optionB = dto.OptionB,
        optionC = dto.OptionC, // FIXED: was OptionB
        optionD = dto.OptionD, // FIXED: was OptionB
        CorrectAnswer = dto.CorrectAnswer.ToUpper(), // FIXED: was missing
        Points = dto.Points
    };

    _context.ModuleQuizQuestions.Add(question);
    await _context.SaveChangesAsync();

    return Ok(new { message = "Quiz question added!" });
}


3. SUBMIT QUIZ FIX (Prevents Foreign Key SQL Error)
---------------------------------------------------
Added module verification and fixed the percentage math (* 100).

[HttpPost("submit-quiz")]
public async Task<IActionResult> SubmitQuiz(SubmitQuizDto dto)
{
    var enrollment = await _context.Enrollments.FindAsync(dto.EnrollmentId);
    if (enrollment == null)
        return NotFound(new { message = "Enrollment not found" });

    // FIX: Verify Module existence to prevent SqlException Foreign Key conflict
    var moduleExists = await _context.CourseModules.AnyAsync(m => m.Id == dto.ModuleId);
    if (!moduleExists)
        return NotFound(new { message = "The module ID provided does not exist in the database." });

    var existingPass = await _context.QuizAttempts
        .AnyAsync(a => a.EnrollmentId == dto.EnrollmentId && a.ModuleId == dto.ModuleId && a.IsPassed);
    if (existingPass)
        return BadRequest(new { message = "You already passed this module" });

    var questions = await _context.ModuleQuizQuestions
        .Where(q => q.ModuleId == dto.ModuleId)
        .ToListAsync();

    if (questions == null || !questions.Any())
        return NotFound(new { message = "No quiz questions found for this module." });

    int score = 0;
    int totalPoints = questions.Sum(q => q.Points);

    // Calculate score
    foreach (var q in questions)
    {
        var submitted = dto.Answers.FirstOrDefault(a => a.QuestionId == q.Id);
        var answerGiven = submitted?.AnswerGiven.ToUpper() ?? "";
        if (answerGiven == q.CorrectAnswer) score += q.Points;
    }

    // FIX: Multiply by 100 for actual percentage integer
    var percentage = (int)Math.Round((double)score / totalPoints * 100);
    var isPassed = percentage >= 70;

    var attempt = new QuizAttempt
    {
        EnrollmentId = dto.EnrollmentId,
        ModuleId = dto.ModuleId, 
        Score = score,
        TotalPoints = totalPoints,
        IsPassed = isPassed,
        AttemptedAt = DateTime.UtcNow
    };

    _context.QuizAttempts.Add(attempt);
    await _context.SaveChangesAsync();

    if (isPassed)
        await RecalculateProgress(dto.EnrollmentId);

    return Ok(new
    {
        score,
        totalPoints,
        percentage,
        isPassed,
        passingScore = "70%",
        message = isPassed ? "Congratulations! Quiz passed!" : "Quiz failed. You may retake it."
    });
}
